dotnet aspnet-codegenerator
dotnet tool install --global dotnet-aspnet-codegenerator --version 9.0.0

dotnet aspnet-codegenerator controller -name RegioesController --model Regiao --dataContext AppDbContext --layout "~/Views/Shared/_LayoutAdmin.cshtml" -outDir Controllers
